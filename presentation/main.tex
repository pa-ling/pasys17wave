#pragma %*\color{blue}{omp parallel for}*)
for (int i = 1; i < arrLen-1; i++)
{ 
    newval[i] = 2*values[i] - oldval[i] + 
    c*(values[i-1] - 2*values[i] + values[i+1]);
}

#pragma %*\color{blue}{omp parallel for}*)
for (int i = 1; i < arrLen-1; i++)
{ 
    oldval[i] = values[i];
    values[i] = newval[i];
}
\end{lstlisting}
